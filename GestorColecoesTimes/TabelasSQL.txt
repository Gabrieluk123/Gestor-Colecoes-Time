CREATE TABLE Times
(
    id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    nome VARCHAR(80) NOT NULL,
    ligaTime VARCHAR(80) NOT NULL,
    sigla VARCHAR(4) NOT NULL
);

CREATE TABLE Camisas
(
    id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    timeId INT NOT NULL,
    nome VARCHAR(80) NOT NULL,
    marca VARCHAR(20) NOT NULL,
    temporada VARCHAR(6) NOT NULL,
    tamanho VARCHAR(5) NOT NULL,
    possui BOOLEAN DEFAULT FALSE NOT NULL,
    localImagem VARCHAR(255) NOT NULL,
    CONSTRAINT FK_TIME FOREIGN KEY (timeId) REFERENCES TIMES(id)
);
